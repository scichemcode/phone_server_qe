Overview & Architecture

Prerequisites

Phone Requirements

  • Android 7.0+ device
  • At least 4GB RAM (6GB+ recommended)
  • At least 10GB free storage
  • Termux installed from F-Droid (not Play Store)

Install from F-Droid

Download and install these apps from f-droid.org:

  • Termux — main terminal app
  • Termux:Boot — for auto-start on reboot
⚠️

Important: Do not install Termux from the Play Store — it is outdated. Always use the F-Droid version.

Battery Optimization (Critical)

Go to Settings → Apps → Termux → Battery → Unrestricted
Do the same for Termux:Boot. This prevents Android from killing the server.

# Test Redis

# Should return: PONG
# Create project directory
mkdir ~/jobserver && cd ~/jobserver

# Create tasks.py
cat > tasks.py << 'EOF'
import time, subprocess

def sleep_job(seconds):
    time.sleep(seconds)
    return f"Slept {seconds}s"

def run_script(script_path, args=None):
    cmd = ["bash", script_path] + (args or [])
    result = subprocess.run(cmd, capture_output=True, text=True)
    return {"stdout": result.stdout, "stderr": result.stderr, "code": result.returncode}

def run_qe(input_file, output_file=None):
    if output_file is None:
        output_file = input_file.replace('.in', '.out')
    pw = '/data/data/com.termux/files/home/q-e/bin/pw.x'
    result = subprocess.run(
        [pw, '-i', input_file],
        stdout=open(output_file, 'w'),
        stderr=subprocess.PIPE,
        text=True
    )
    return {
        "output_file": output_file,
        "returncode": result.returncode,
        "stderr": result.stderr
    }
EOF

Step 3: Flask REST API

📱 On Phone (via SSH) 
# Create server.py using Python
python3 -c "
content = '''from flask import Flask, request, jsonify, send_file
from rq import Queue
from rq.job import Job
from redis import Redis
import tasks, os

app = Flask(__name__)
conn = Redis()
q = Queue(connection=conn)
UPLOAD_DIR = \"/data/data/com.termux/files/home/qe_jobs\"
os.makedirs(UPLOAD_DIR, exist_ok=True)

@app.route(\"/upload\", methods=[\"POST\"])
def upload():
    if \"file\" not in request.files:
        return jsonify({\"error\": \"no file provided\"}), 400
    f = request.files[\"file\"]
    path = os.path.join(UPLOAD_DIR, f.filename)
    f.save(path)
    return jsonify({\"path\": path, \"message\": \"uploaded successfully\"})

@app.route(\"/submit\", methods=[\"POST\"])
def submit():
    data = request.json
    job_type = data.get(\"type\")
    if job_type == \"qe\":
        job = q.enqueue(tasks.run_qe, data[\"input\"], data.get(\"output\"), job_timeout=3600)
    elif job_type == \"sleep\":
        job = q.enqueue(tasks.sleep_job, data[\"seconds\"])
    else:
        return jsonify({\"error\": \"unknown job type\"}), 400
    return jsonify({\"job_id\": job.id, \"status\": str(job.get_status())})

@app.route(\"/status/\", methods=[\"GET\"])
def status(job_id):
    job = Job.fetch(job_id, connection=conn)
    return jsonify({\"id\": job.id, \"status\": str(job.get_status()), \"result\": job.result})

@app.route(\"/jobs\", methods=[\"GET\"])
def list_jobs():
    queued = [j.id for j in q.get_jobs()]
    return jsonify({\"queued\": queued, \"length\": len(q)})

@app.route(\"/download\", methods=[\"GET\"])
def download():
    path = request.args.get(\"path\")
    if not path:
        return jsonify({\"error\": \"no path provided\"}), 400
    try:
        return send_file(path, as_attachment=True)
    except Exception as e:
        return jsonify({\"error\": str(e)}), 404

if __name__ == \"__main__\":
    app.run(host=\"0.0.0.0\", port=5000)
'''
open('/data/data/com.termux/files/home/jobserver/server.py', 'w').write(content)
print('done')
"

Step 4: Compile Quantum Espresso

⏱️

Compilation takes 30–60 minutes. Keep the phone plugged in and make sure Termux stays active.

📱 On Phone (via SSH) 
# Install build dependencies
pkg install cmake make clang flang openblas fftw patchelf llvm -y

# Install missing library fix
pkg install libandroid-complex-math -y
ln -s /data/data/com.termux/files/usr/lib/libandroid-complex-math.so \
      /data/data/com.termux/files/usr/lib/libandroid-complex-math.a

# Create symlinks for missing tools
ln -s /data/data/com.termux/files/usr/bin/llvm-ar \
      /data/data/com.termux/files/usr/bin/ar
ln -s /data/data/com.termux/files/usr/bin/flang \
      /data/data/com.termux/files/usr/bin/gfortran

# Clone Quantum Espresso
cd ~ && git clone https://github.com/QEF/q-e.git
cd ~/q-e
# Configure with flang
./configure F77=flang FC=flang

# Fix make.inc (remove duplicate compiler entries)
grep "^MPIF90\|^F90\|^LD" make.inc
# Ensure these are all set to 'flang' with no duplicates:
# MPIF90 = flang
# F90    = flang
# LD     = flang

# Fix lapack make.inc
sed -i 's/FORTRAN  = gfortran/FORTRAN  = flang/' ~/q-e/install/make_lapack.inc
sed -i 's/OPTS     = -O3 -g/OPTS     =/' ~/q-e/install/make_lapack.inc
sed -i 's/LOADOPTS = -g/LOADOPTS =/' ~/q-e/install/make_lapack.inc
# Build pw.x with all cores
make pw -j6 2>&1 | tail -20

# Verify binary exists
ls ~/q-e/bin/pw.x
# Download Silicon pseudopotential
mkdir -p ~/pseudo ~/out ~/qe_jobs
wget https://pseudopotentials.quantum-espresso.org/upf_files/Si.pbe-n-rrkjus_psl.1.0.0.UPF \
     -P ~/pseudo/

Step 5: Full Job Submission Workflow

Save this as submit.py on your laptop:

💻 On Laptop 
import requests, sys, time, os

PHONE_URL = "http://100.83.65.14:5000"  # Your phone's Tailscale IP

def upload_and_submit(input_file):
    print(f"Uploading {input_file}...")
    with open(input_file, 'rb') as f:
        r = requests.post(f"{PHONE_URL}/upload", files={"file": f})
    remote_path = r.json()["path"]
    print(f"Uploaded to: {remote_path}")

    r = requests.post(f"{PHONE_URL}/submit", json={
        "type": "qe", "input": remote_path
    })
    job_id = r.json()["job_id"]
    print(f"Job ID: {job_id}")

    while True:
        r = requests.get(f"{PHONE_URL}/status/{job_id}")
        status = r.json()["status"]
        print(f"Status: {status}")
        if "FINISHED" in status:
            return job_id, r.json()["result"]["output_file"]
        elif "FAILED" in status:
            sys.exit("Job failed!")
        time.sleep(5)

def download_output(output_file):
    local_file = os.path.basename(output_file)
    r = requests.get(f"{PHONE_URL}/download", params={"path": output_file})
    open(local_file, 'wb').write(r.content)
    print(f"Saved to {local_file}")

if __name__ == "__main__":
    job_id, output = upload_and_submit(sys.argv[1])
    download_output(output)
    print("Done!")
# Usage
python3 submit.py my_calculation.in
1
Upload POST /upload — sends .in file to phone's ~/qe_jobs/
2
Submit POST /submit — enqueues job in Redis
3
Execute RQ worker picks up job, runs pw.x
4
Poll GET /status — checks every 5s until FINISHED
5
Download GET /download — retrieves output file to laptop

Step 6: Auto-Start on Boot

Install Termux:Boot from F-Droid, open it once, then create the boot script:

📱 On Phone (via SSH) 
mkdir -p ~/.termux/boot

cat > ~/.termux/boot/start_server.sh << 'EOF'
#!/data/data/com.termux/files/usr/bin/bash
termux-wake-lock
sleep 10
pkill -f "redis-server"
pkill -f "rq worker"
pkill -f "python server.py"
sleep 2
redis-server --daemonize yes
sleep 2
cd /data/data/com.termux/files/home/jobserver
rq worker &
sleep 2
python server.py &
EOF

chmod +x ~/.termux/boot/start_server.sh
ℹ️

The sleep 10 gives Android time to fully boot before starting services. After phone restart, the server will be available in ~15 seconds.

Step 7: Remote Access via Tailscale

Tailscale creates a secure VPN mesh so you can access your phone from anywhere without port forwarding.

💻 On Laptop 
# Install Tailscale
curl -fsSL https://tailscale.com/install.sh | sh
sudo tailscale up
# Visit the URL shown to authenticate

# Get your laptop's Tailscale IP
tailscale ip
📱 On Phone

Install Tailscale from the Play Store and log in with the same account.

💻 On Laptop — Test Remote Access 
# Replace with your phone's Tailscale IP (shown in Tailscale app)
curl http://100.83.65.14:5000/jobs
# Should return: {"length":0,"queued":[]}
🎉

Once connected, update PHONE_URL in submit.py to your phone's Tailscale IP. You can now submit jobs from anywhere in the world!
